CREATE VIEW WykazProduktow AS
(SELECT ProductID, Count(ProductID) AS IloscZamowionych FROM OrderDetails
GROUP BY ProductID);

INNER JOIN WykazProduktow ON Products.ProductID = WykazProduktow.ProductID;
